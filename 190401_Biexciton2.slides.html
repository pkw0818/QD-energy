vh = spsolve(vcoul, np.abs(b.psi_h2)*a.e/a.eo)

veDiag = np.diag(ve)
vhDiag = np.diag(vh)

Jee = np.dot(b.psi_e2.T, np.dot(veDiag, b.psi_e2))
Jhh = np.dot(b.psi_h2.T, np.dot(vhDiag, b.psi_h2))

braketH = np.dot(b.psi_h2.T, b.psi_h2)

Jhh /= braketH; #Jhh = Jhh[0]
